# ------------------------------------------------------------

# ------------------------------------------------------------

# Web framework — handles all HTTP endpoints, request validation,
# serves HTML pages, and auto-generates /docs Swagger UI
fastapi>=0.115.0

# ASGI server — actually runs the FastAPI app
# Use:  uvicorn main:app --reload
# The [standard] extra includes websockets + watchfiles for --reload
uvicorn[standard]>=0.30.0

# Data validation — validates every API request and response shape
# Pydantic v2 is required (v1 will NOT work — different API)
pydantic>=2.5.0

# Encryption and password hashing — required for:
#   - Fernet encryption of stored score data (AES-128-CBC + HMAC)
#   - PBKDF2-HMAC-SHA256 password hashing (260,000 iterations)
cryptography>=42.0.0

# .env file loading — reads SCORE_ENCRYPTION_KEY, GEMINI_API_KEY,
# GROQ_API_KEY from the .env file into environment variables
python-dotenv>=1.0.0


# ------------------------------------------------------------
# AI FEATURES — Optional but strongly recommended.

# ------------------------------------------------------------
